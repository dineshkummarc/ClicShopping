version varchar(50) not_null comment(Version of ground truth data)
created_at datetime not_null comment(When ground truth was created)
updated_at datetime DEFAULT NULL comment(When ground truth was last updated)
notes text DEFAULT NULL comment(Additional notes about this ground truth)
