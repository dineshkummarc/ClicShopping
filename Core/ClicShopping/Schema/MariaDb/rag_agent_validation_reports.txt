id int(11) not_null auto_increment
report_id varchar(255) not_null comment(Unique identifier for the report)
execution_time datetime not_null comment(When the validation suite was executed)
total_tests int(11) not_null comment(Total number of tests executed)
passed_tests int(11) not_null comment(Number of tests that passed)
failed_tests int(11) not_null comment(Number of tests that failed)
skipped_tests int(11) not_null comment(Number of tests that were skipped)
error_tests int(11) not_null comment(Number of tests with execution errors)
critical_inconsistencies int(11) not_null comment(Number of critical inconsistencies detected)
summary longtext CHARACTER SET utf8mb4 COLLATE utf8mb4_bin not_null CHECK (json_valid(`summary`)) comment(Summary statistics by complexity and mode)
performance_analysis longtext CHARACTER SET utf8mb4 COLLATE utf8mb4_bin default(NULL) CHECK (json_valid(`performance_analysis`)) comment(Performance metrics analysis)
cache_impact_analysis longtext CHARACTER SET utf8mb4 COLLATE utf8mb4_bin default(NULL) CHECK (json_valid(`cache_impact_analysis`)) comment(Cache impact analysis results)
memory_validation_results longtext CHARACTER SET utf8mb4 COLLATE utf8mb4_bin default(NULL) CHECK (json_valid(`memory_validation_results`)) comment(Memory context validation results)
multi_factor_analysis longtext CHARACTER SET utf8mb4 COLLATE utf8mb4_bin default(NULL) CHECK (json_valid(`multi_factor_analysis`)) comment(Multi-factor prompt impact analysis)
report_file_path varchar(500) default(NULL) comment(Path to detailed report file)
created_at timestamp NULL default(current_timestamp)

unique report_id report_id
idx_execution_time execution_time
idx_report_id report_id
idx_created created_at
