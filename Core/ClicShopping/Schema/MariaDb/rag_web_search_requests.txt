id INT(11) not_null auto_increment
query TEXT not_null comment(Original user query)
translated_query TEXT comment(Query translated into English)
intent ENUM('INTERNAL_RAG','EXTERNAL_WEB','ANALYTICS','MIXED') not_null default('EXTERNAL_WEB') comment(Classified intent)
intent_confidence FLOAT default(0.0) comment(Classification confidence (0–1))
search_engine VARCHAR(50) default('serpapi') comment(Search engine used (serpapi, bing, google))
results_count INT default(0) comment(Number of results returned)
response_summary TEXT comment(Summary generated by the LLM)
quality_score FLOAT default(0.0) comment(Quality score of the results (0–1))
cached TINYINT(1) default(0) comment(if the result comes from cache)
cached_in_rag TINYINT(1) default(0) comment(if stored in rag_web_cache_embedding for learning)
execution_time FLOAT default(0.0) comment(Execution time in seconds)
api_cost FLOAT default(0.0) comment(Estimated API cost in USD)
user_admin VARCHAR(255) comment(User who performed the search)
session_id VARCHAR(255) comment(Session ID)
ip_address VARCHAR(45) comment(IP address (IPv4 or IPv6))
audit_data JSON comment(Complete audit data (JSON))
date_added DATETIME default(CURRENT_TIMESTAMP) comment(Creation date)
