agent_id varchar(100) not_null comment(Unique agent identifier)
enabled tinyint(1) not_null DEFAULT 1 comment(Agent enabled state (1=enabled, 0=disabled))
agent_type enum('actor','critic','hybrid') not_null comment(Agent type)
domain varchar(50) DEFAULT NULL comment(Domain specialization ecommerce, hr, finance, trading, etc)
description text DEFAULT NULL comment(Agent description)
created_at datetime not_null comment(Registration timestamp)
updated_at datetime not_null comment(Last update timestamp)

idx_enabled enabled
idx_agent_type agent_type
idx_domain domain
idx_enabled_type enabled agent_type
idx_enabled_domain enabled domain
