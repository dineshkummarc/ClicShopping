# Semantic Search Orchestrator Language Definitions
# Patterns pour détecter les réponses génériques "informations insuffisantes"
# NOTE: Tout le traitement se fait en anglais, donc seuls les patterns anglais sont nécessaires

# Patterns de détection d'informations insuffisantes (séparés par des pipes)
text_insufficient_info_patterns = I don't have that information|I don't have enough information|I would need more specific information|consult the company's official website|contact their customer service|don't have enough information|no information found|I don't have|insufficient information|Informations Insuffisantes|Je n'ai pas cette information|Je n'ai pas assez d'informations|n'ai pas assez d'informations|J'aurais besoin de plus d'informations|aucune information

# Longueur minimale de réponse pour être considérée comme valide (non générique)
text_min_response_length = 20

# Messages de log
text_log_conversation_memory_useful = Le document de mémoire de conversation a un contenu utile (longueur: {{length}})
text_log_conversation_memory_generic = Le document de mémoire de conversation a une réponse LLM générique - ignoré
text_log_conversation_memory_no_content = Les documents de mémoire de conversation n'ont pas de contenu réel (vides ou réponses génériques)
text_log_extracted_response = Réponse utile extraite de la mémoire de conversation (longueur: {{length}})
text_log_using_extracted_response = Utilisation de la réponse extraite de la mémoire de conversation directement
text_log_could_not_extract = Impossible d'extraire une réponse utile de la mémoire de conversation
text_log_rag_generic_response = RAG a retourné une réponse générique 'pas d'information' - documents trouvés mais non pertinents - déclenchement du LLM fallback
text_log_rag_synthesis_failed = Échec de la synthèse RAG: {{error}} - retour des résultats bruts
text_log_document_stores_no_results = Recherche DocumentStores: aucun résultat
text_log_conversation_memory_no_results = Recherche ConversationMemory: aucun résultat
text_log_conversation_memory_empty = La mémoire de conversation a retourné des résultats vides ou sans contenu - passage au fallback suivant
